POLYFILL FOR CALL FUNCTIONS

Function.prototype.call = function(context,...args){
    const fn = this;

    context.fn = fn;
    context.fn(...args);
};

function buyCar(price,color){
console.log(`I have car ${this.name} of price ${price}
and ${color}`);
}


const jaguar = {
    name :"02-khushi"
}


buyCar.call(jaguar,2000,"blue");
